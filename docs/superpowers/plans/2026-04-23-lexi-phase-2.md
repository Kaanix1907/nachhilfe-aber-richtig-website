```typescript
import { toZonedTime, fromZonedTime } from "date-fns-tz";
import { addDays, startOfDay, differenceInSeconds } from "date-fns";

const TZ = "Europe/Berlin";

export function secondsUntilBerlinMidnight(now: Date = new Date()): number {
  const berlinNow = toZonedTime(now, TZ);
  const berlinMidnight = addDays(startOfDay(berlinNow), 1);
  const utcMidnight = fromZonedTime(berlinMidnight, TZ);
  return Math.max(1, differenceInSeconds(utcMidnight, now));
}
```

```typescript
import { Redis } from "@upstash/redis";
import { secondsUntilBerlinMidnight } from "./berlin-time";

export const USER_DAILY_LIMIT = 10;
export const ANON_LIFETIME_LIMIT = 2;

const redis = new Redis({
  url: process.env.UPSTASH_REDIS_REST_URL!,
  token: process.env.UPSTASH_REDIS_REST_TOKEN!,
});

interface LimitResult {
  allowed: boolean;
  used: number;
  limit: number;
  resetAt: Date | null;
}

function userKey(userId: string): string {
  return `lexi:rl:user:${userId}`;
}

function anonKey(anonId: string): string {
  return `lexi:rl:anon:${anonId}`;
}

function anonIpKey(ip: string): string {
  return `lexi:rl:anon-ip:${ip}`;
}

export async function checkUserLimit(userId: string): Promise<LimitResult> {
  const key = userKey(userId);
  const count = (await redis.incr(key)) as number;

  // Beim ersten Treffer TTL auf Berlin-Midnight setzen
  if (count === 1) {
    const ttl = secondsUntilBerlinMidnight();
    await redis.pexpire(key, ttl * 1000);
  }

  const resetAt = new Date(Date.now() + secondsUntilBerlinMidnight() * 1000);

  return {
    allowed: count <= USER_DAILY_LIMIT,
    used: count,
    limit: USER_DAILY_LIMIT,
    resetAt,
  };
}

export async function checkAnonLimit(anonId: string): Promise<LimitResult> {
  const key = anonKey(anonId);
  const count = (await redis.incr(key)) as number;
  // Kein TTL — anonymes Limit ist lifetime

  return {
    allowed: count <= ANON_LIFETIME_LIMIT,
    used: count,
    limit: ANON_LIFETIME_LIMIT,
    resetAt: null,
  };
}

export async function checkAnonIpLimit(ip: string): Promise<LimitResult> {
  const key = anonIpKey(ip);
  const count = (await redis.incr(key)) as number;
  if (count === 1) {
    const ttl = secondsUntilBerlinMidnight();
    await redis.pexpire(key, ttl * 1000);
  }
  const resetAt = new Date(Date.now() + secondsUntilBerlinMidnight() * 1000);
  return { allowed: count <= 20, used: count, limit: 20, resetAt };
}

export async function getUserUsage(userId: string): Promise<LimitResult> {
  const count = ((await redis.get(userKey(userId))) as number | null) ?? 0;
  const resetAt = new Date(Date.now() + secondsUntilBerlinMidnight() * 1000);
  return { allowed: count < USER_DAILY_LIMIT, used: count, limit: USER_DAILY_LIMIT, resetAt };
}

export async function getAnonUsage(anonId: string): Promise<LimitResult> {
  const count = ((await redis.get(anonKey(anonId))) as number | null) ?? 0;
  return { allowed: count < ANON_LIFETIME_LIMIT, used: count, limit: ANON_LIFETIME_LIMIT, resetAt: null };
}
```

```typescript
import { clerkMiddleware, createRouteMatcher } from "@clerk/nextjs/server";
import { NextResponse } from "next/server";
import { randomUUID } from "node:crypto";
import { ANON_COOKIE_NAME, ANON_COOKIE_MAX_AGE, isValidAnonId } from "@/lib/anon-cookie";

// Nur /api/upload und /api/migrate-anon MÜSSEN eingeloggt sein
const isProtectedRoute = createRouteMatcher(["/api/upload(.*)", "/api/migrate-anon(.*)"]);

// Routen die auch anonym funktionieren aber den Cookie brauchen
const needsAnonCookie = createRouteMatcher(["/chat(.*)", "/api/chat(.*)", "/api/chats(.*)", "/api/usage(.*)"]);

export default clerkMiddleware(async (auth, req) => {
  if (isProtectedRoute(req)) await auth.protect();

  const res = NextResponse.next();

  if (needsAnonCookie(req)) {
    const { userId } = await auth();
    const existing = req.cookies.get(ANON_COOKIE_NAME)?.value;

    if (!userId && (!existing || !isValidAnonId(existing))) {
      res.cookies.set(ANON_COOKIE_NAME, randomUUID(), {
        httpOnly: true,
        sameSite: "lax",
        path: "/",
        maxAge: ANON_COOKIE_MAX_AGE,
      });
    }
  }

  return res;
});

export const config = {
  matcher: [
    "/((?!_next|[^?]*\\.(?:html?|css|js(?!on)|jpe?g|webp|png|gif|svg|ttf|woff2?|ico|csv|docx?|xlsx?|zip|webmanifest)).*)",
    "/(api|trpc)(.*)",
  ],
};
```

```typescript
import { auth } from "@clerk/nextjs/server";

import { db } from "@/db/client";
import { chats, messages, anonChats, anonMessages } from "@/db/schema";

import { checkUserLimit, checkAnonLimit, checkAnonIpLimit } from "@/lib/rate-limit";
import { ANON_COOKIE_NAME, isValidAnonId } from "@/lib/anon-cookie";

export const maxDuration = 60;

interface ChatRequestBody {
  chatId?: string;
  subject?: string;
  message: string;
  attachmentUrl?: string;
  attachmentContentType?: string;
}

export async function POST(req: Request) {
  const body = (await req.json()) as ChatRequestBody;
  const { chatId: providedChatId, subject, message, attachmentUrl } = body;

  // Validation
  if ((!message || message.trim().length === 0) && !attachmentUrl) {
    return new Response("Empty message", { status: 400 });
  }
  if (message && message.length > 2000) {
    return new Response("Message too long (max 2000 chars)", { status: 400 });
  }

  // Auth resolution: user oder anon
  const { userId } = await auth();
  const cookieStore = await cookies();
  const anonIdRaw = cookieStore.get(ANON_COOKIE_NAME)?.value;
  const anonId = userId ? null : anonIdRaw && isValidAnonId(anonIdRaw) ? anonIdRaw : null;

  if (!userId && !anonId) {
    return new Response("cookie_missing", { status: 401 });
  }

  // Anonyme User dürfen keine Uploads
  if (anonId && attachmentUrl) {
    return new Response("upload_requires_auth", { status: 403 });
  }

  // Rate Limit
  if (userId) {
    const limit = await checkUserLimit(userId);
    if (!limit.allowed) {
      return new Response("daily_limit_reached", {
        status: 429,
        headers: {
          "X-Usage-Used": String(limit.used),
          "X-Usage-Limit": String(limit.limit),
          "X-Usage-Reset-At": limit.resetAt?.toISOString() ?? "",
          "X-Usage-Mode": "user",
        },
      });
    }
  } else if (anonId) {
    // IP-Second-Limit
    const hdrs = await headers();
    const ip = hdrs.get("x-forwarded-for")?.split(",")[0].trim() || "unknown";
    const ipLimit = await checkAnonIpLimit(ip);
    if (!ipLimit.allowed) {
      return new Response("ip_limit_reached", { status: 429 });
    }

    const limit = await checkAnonLimit(anonId);
    if (!limit.allowed) {
      return new Response("anon_limit_reached", {
        status: 429,
        headers: {
          "X-Usage-Used": String(limit.used),
          "X-Usage-Limit": String(limit.limit),
          "X-Usage-Mode": "anon",
        },
      });
    }
  }

  // Chat erstellen/holen + Message speichern
  let finalChatId: string;
  let modelMessages: ModelMessage[];
  let useAdvanced = false;

  if (userId) {
    finalChatId = providedChatId ?? "";
    if (!finalChatId) {
      const title = message.trim().slice(0, 60) || "Bildfrage";
      const [newChat] = await db.insert(chats).values({ userId, title, subject }).returning();
      finalChatId = newChat.id;
    }

    await db.insert(messages).values({
      chatId: finalChatId,
      role: "user",
      content: message,
      attachmentUrl: attachmentUrl ?? null,
    });

    const history = await db.query.messages.findMany({
      where: eq(messages.chatId, finalChatId),
      orderBy: (m, { asc }) => [asc(m.createdAt)],
    });

    modelMessages = history.map((m) => {
      if (m.role === "user" && m.attachmentUrl) {
        return {
          role: "user",
          content: [
            { type: "image", image: new URL(m.attachmentUrl) },
            { type: "text", text: m.content || "Bitte hilf mir bei dieser Aufgabe aus dem Bild." },
          ],
        };
      }
      return { role: m.role as "user" | "assistant", content: m.content };
    });

    useAdvanced = history.some((m) => m.attachmentUrl !== null && m.attachmentUrl !== "");
  } else {
    finalChatId = providedChatId ?? "";
    if (!finalChatId) {
      const title = message.trim().slice(0, 60);
      const [newChat] = await db
        .insert(anonChats)
        .values({ anonId: anonId!, title, subject })
        .returning();
      finalChatId = newChat.id;
    }

    await db.insert(anonMessages).values({
      chatId: finalChatId,
      role: "user",
      content: message,
    });

    const history = await db
      .select({ role: anonMessages.role, content: anonMessages.content })
      .from(anonMessages)
      .where(eq(anonMessages.chatId, finalChatId))
      .orderBy(anonMessages.createdAt);

    modelMessages = history.map((m) => ({
      role: m.role as "user" | "assistant",
      content: m.content,
    }));
  }

  const selectedModel = useAdvanced ? models.advanced : models.default;

  const result = streamText({
    model: selectedModel,
    system: buildSystemPrompt({ subject }),
    messages: modelMessages,
    onFinish: async ({ text }) => {
      if (userId) {
        await db.insert(messages).values({ chatId: finalChatId, role: "assistant", content: text });
        await db.update(chats).set({ updatedAt: new Date() }).where(eq(chats.id, finalChatId));
      } else {
        await db.insert(anonMessages).values({ chatId: finalChatId, role: "assistant", content: text });
        await db.update(anonChats).set({ updatedAt: new Date() }).where(eq(anonChats.id, finalChatId));
      }
    },
  });

  // Usage-Header anhängen für Frontend
  const usageHeaders: Record<string, string> = { "X-Chat-Id": finalChatId };
  if (userId) {
    const { checkUserLimit: _, getUserUsage } = await import("@/lib/rate-limit");
    const usage = await getUserUsage(userId);
    usageHeaders["X-Usage-Used"] = String(usage.used);
    usageHeaders["X-Usage-Limit"] = String(usage.limit);
    usageHeaders["X-Usage-Reset-At"] = usage.resetAt?.toISOString() ?? "";
    usageHeaders["X-Usage-Mode"] = "user";
  } else if (anonId) {
    const { getAnonUsage } = await import("@/lib/rate-limit");
    const usage = await getAnonUsage(anonId);
    usageHeaders["X-Usage-Used"] = String(usage.used);
    usageHeaders["X-Usage-Limit"] = String(usage.limit);
    usageHeaders["X-Usage-Mode"] = "anon";
  }

  return result.toTextStreamResponse({ headers: usageHeaders });
}
```

```typescript
import { db } from "@/db/client";
import { chats, messages, anonChats, anonMessages } from "@/db/schema";
import { eq } from "drizzle-orm";

export async function migrateAnonChats(
  anonId: string,
  userId: string,
): Promise<{ migrated: number }> {
  return await db.transaction(async (tx) => {
    const existing = await tx
      .select()
      .from(anonChats)
      .where(eq(anonChats.anonId, anonId));

    if (existing.length === 0) return { migrated: 0 };

    for (const old of existing) {
      const [newChat] = await tx
        .insert(chats)
        .values({
          userId,
          title: old.title,
          subject: old.subject,
          createdAt: old.createdAt,
          updatedAt: old.updatedAt,
        })
        .returning();

      const oldMessages = await tx
        .select()
        .from(anonMessages)
        .where(eq(anonMessages.chatId, old.id));

      if (oldMessages.length > 0) {
        await tx.insert(messages).values(
          oldMessages.map((m) => ({
            chatId: newChat.id,
            role: m.role,
            content: m.content,
            tokenCount: m.tokenCount,
            createdAt: m.createdAt,
          })),
        );
      }

      await tx.delete(anonChats).where(eq(anonChats.id, old.id));
    }

    return { migrated: existing.length };
  });
}
```

```typescript
"use client";

import { useEffect, useRef } from "react";
import { useUser } from "@clerk/nextjs";

/**
 * Wenn ein vorher anonymer User sich einloggt, triggert dieser Hook
 * einmalig die Migration von anon_chats → chats.
 */
export function useAnonMigration() {
  const { user, isLoaded } = useUser();
  const migratedRef = useRef(false);

  useEffect(() => {
    if (!isLoaded || !user || migratedRef.current) return;

    migratedRef.current = true;
    (async () => {
      try {
        const res = await fetch("/api/migrate-anon", { method: "POST" });
        if (!res.ok) return;
        const data = (await res.json()) as { migrated: number };
        if (data.migrated > 0) {
          // Reload, damit Chat-Liste aktuell ist
          window.location.reload();
        }
      } catch {
        // Silent: Fehler bei Migration ist nicht kritisch
      }
    })();
  }, [isLoaded, user]);
}
```

```typescript
async function send() {
  if ((!input.trim() && !pendingAttachment) || isLoading) return;
  if (rateLimited) return;

  const text = input.trim();
  const attachment = pendingAttachment;
  const userMsg: MessageProps = { role: "user", content: text, attachmentUrl: attachment?.url };

  setMessages((prev) => [...prev, userMsg, { role: "assistant", content: "" }]);
  setInput("");
  setPendingAttachment(null);
  setIsLoading(true);

  const res = await fetch("/api/chat", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      chatId,
      message: text,
      attachmentUrl: attachment?.url,
      attachmentContentType: attachment?.contentType,
    }),
  });

  // Rate-Limit-Response?
  if (res.status === 429) {
    const errText = await res.text();
    setMessages((prev) => prev.slice(0, -2)); // remove optimistic user+assistant
    setIsLoading(false);

    if (errText === "anon_limit_reached") {
      setShowAnonGate(true);
    } else {
      setRateLimited(true);
    }
    refreshUsage();
    return;
  }

  if (!res.ok || !res.body) {
    setMessages((prev) => [
      ...prev.slice(0, -1),
      { role: "assistant", content: "Oops, Lexi hat kurz einen Wackler. Versuch's bitte nochmal." },
    ]);
    setIsLoading(false);
    return;
  }

  const newChatId = res.headers.get("X-Chat-Id");
  if (newChatId) setChatId(newChatId);

  // Usage-Header sofort auslesen
  const used = res.headers.get("X-Usage-Used");
  const limit = res.headers.get("X-Usage-Limit");
  const mode = res.headers.get("X-Usage-Mode") as "user" | "anon" | null;
  const resetAt = res.headers.get("X-Usage-Reset-At");
  if (used && limit && mode) {
    setUsage({ used: Number(used), limit: Number(limit), mode, resetAt });
  }

  const reader = res.body.getReader();
  const decoder = new TextDecoder();
  let assistantText = "";
  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    assistantText += decoder.decode(value, { stream: true });
    setMessages((prev) => [...prev.slice(0, -1), { role: "assistant", content: assistantText }]);
  }
  setIsLoading(false);
  refreshChatList();

  // Nach Stream-Ende: Anon-Gate wenn 2/2
  if (mode === "anon" && used === limit) {
    setShowAnonGate(true);
  }
}
```
